```bash
#!/bin/bash
# Собираем все KB-файлы в один markdown для импорта
OUTPUT="data/kb-export-$(date +%Y%m%d).md"

echo "# Shadow Stack Knowledge Base Export" > "$OUTPUT"
echo "Generated: $(date)" >> "$OUTPUT"
echo "" >> "$OUTPUT"

# Ключевые файлы
for f in CLAUDE.md AGENTS.md ORCHESTRATOR.md RECOMMENDATIONS.md; do
  [ -f "$f" ] && echo -e "\n---\n# $f\n" >> "$OUTPUT" && cat "$f" >> "$OUTPUT"

AF_KB="$HOME/agent-factory/.agent/knowledge/agent-factory-kb.md"
[ -f "$AF_KB" ] && echo -e "\n---\n# Agent Factory KB\n" >> "$OUTPUT" && cat "$AF_KB" >> "$OUTPUT"

echo "✅ Exported to $OUTPUT ($(wc -c < "$OUTPUT") bytes)"
echo "Import to NotebookLM: https://notebooklm.google.com/"
```

### D.7.2: Обновить server/lib/supabase.cjs (CJS!)

Файл: `~/shadow-stack_local_1/server/lib/supabase.cjs` (НОВЫЙ или обновить существующий)

```javascript
const { createClient } = require('@supabase/supabase-js');
const config = require('./config.cjs');

const supabase = createClient(
  config.SUPABASE_URL,
  config.SUPABASE_KEY
);

async function logRouterRequest(data) {
  try {
    const { error } = await supabase
      .from('logs')  // НЕ router_logs!
      .insert([{
        prompt: data.prompt,
        tier: data.tier || 'local',
        model: data.model,
        latency: Math.round(data.latency_ms || data.latency || 0),
        tokens: data.tokens || 0,
        cost_usd: data.cost_usd || 0,
        quality: data.quality || null,
        error: data.error || null,
        cached: data.cached || false,
        agent_id: data.agent_id || 'orchestrator',
        session_id: data.session_id || null,
        project: 'shadow-stack-prod'
      }]);
    if (error) throw error;
  } catch (err) {
    // Silent fail — не ронять роутер из-за логов
    console.error('[Supabase Log Error]:', err.message);
  }
}

module.exports = { supabase, logRouterRequest };
```

```sql
CREATE TABLE IF NOT EXISTS public.logs (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  timestamp timestamptz DEFAULT now(),
  prompt text,
  tier text,
  model text,
  latency integer,
  tokens integer DEFAULT 0,
```
